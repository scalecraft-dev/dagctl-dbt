To create your first dbt project, create a DbtProject resource:

  apiVersion: orchestration.scalecraft.io/v1alpha1
  kind: DbtProject
  metadata:
    name: my-dbt-project
  spec:
    git:
      repository: https://github.com/your-org/your-dbt-repo.git
      ref: main
    image: scalecraft/dagctl-dbt-demo:latest
    schedule: "0 */6 * * * *"  # Every 6 hours
    commands:
      - run
